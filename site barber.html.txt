            const currentMonthYear = document.getElementById('current-month');
            
            // Set month/year header
            const months = ["Janeiro", "Fevereiro", "Março", "Abril", "Maio", "Junho", "Julho", "Agosto", "Setembro", "Outubro", "Novembro", "Dezembro"];
            currentMonthYear.textContent = `${months[currentDate.getMonth()]} ${currentDate.getFullYear()}`;
            
            // Clear previous days
            calendarDays.innerHTML = '';
            
            // Get first day of month and total days in month
            const firstDay = new Date(currentDate.getFullYear(), currentDate.getMonth(), 1).getDay();
            const daysInMonth = new Date(currentDate.getFullYear(), currentDate.getMonth() + 1, 0).getDate();
            
            // Get days from previous month
            const prevMonthDays = new Date(currentDate.getFullYear(), currentDate.getMonth(), 0).getDate();
            
            // Add days from previous month (disabled)
            for (let i = 0; i < firstDay; i++) {
                const dayElement = document.createElement('div');
                dayElement.className = 'calendar-day text-center py-2 text-gray-400 disabled';
                dayElement.textContent = prevMonthDays - firstDay + i + 1;
                calendarDays.appendChild(dayElement);
            }
            
            // Add days from current month
            const today = new Date();
            for (let i = 1; i <= daysInMonth; i++) {
                const dayElement = document.createElement('div');
                dayElement.className = 'calendar-day text-center py-2 cursor-pointer';
                
                // Highlight today
                if (i === today.getDate() && currentDate.getMonth() === today.getMonth() && currentDate.getFullYear() === today.getFullYear()) {
                    dayElement.classList.add('bg-amber-100', 'font-bold');
                }
                
                // Highlight selected date
                if (selectedDate && i === selectedDate.getDate() && currentDate.getMonth() === selectedDate.getMonth() && currentDate.getFullYear() === selectedDate.getFullYear()) {
                    dayElement.classList.add('bg-amber-500', 'text-white');
                }
                
                dayElement.textContent = i;
                dayElement.dataset.day = i;
                
                // Disable past days
                if (currentDate.getMonth() === today.getMonth() && currentDate.getFullYear() === today.getFullYear() && i < today.getDate()) {
                    dayElement.classList.add('text-gray-400', 'disabled');
                } else {
                    dayElement.addEventListener('click', function() {
                        selectDate(i);
                    });
                }
                
                calendarDays.appendChild(dayElement);
            }
            
            // Calculate days from next month to show (to fill the grid)
            const totalDaysShown = firstDay + daysInMonth;
            const nextMonthDays = totalDaysShown <= 35 ? 35 - totalDaysShown : 42 - totalDaysShown;
            
            // Add days from next month (disabled)
            for (let i = 1; i <= nextMonthDays; i++) {
                const dayElement = document.createElement('div');
                dayElement.className = 'calendar-day text-center py-2 text-gray-400 disabled';
                dayElement.textContent = i;
                calendarDays.appendChild(dayElement);
            }
        }
        
        function selectDate(day) {
            selectedDate = new Date(currentDate.getFullYear(), currentDate.getMonth(), day);
            
            // Update calendar UI
            document.querySelectorAll('.calendar-day').forEach(el => {
                el.classList.remove('bg-amber-500', 'text-white');
            });
            
            const selectedDayElement = document.querySelector(`.calendar-day[data-day="${day}"]:not(.disabled)`);
            if (selectedDayElement) {
                selectedDayElement.classList.add('bg-amber-500', 'text-white');
            }
            
            // Generate time slots
            generateTimeSlots();
            
            // Update booking summary
            updateBookingSummary();
        }
        
        function generateTimeSlots() {
            const timeSlotsContainer = document.getElementById('time-slots');
            timeSlotsContainer.innerHTML = '';
            
            if (!selectedDate) return;
            
            // Generate time slots from 9AM to 6PM, every 30 minutes
            const startHour = 9;
            const endHour = 18;
            const today = new Date();
            
            for (let hour = startHour; hour < endHour; hour++) {
                for (let minute = 0; minute < 60; minute += 30) {
                    // Skip times that have already passed for today
                    if (selectedDate.getDate() === today.getDate() && 
                        selectedDate.getMonth() === today.getMonth() && 
                        selectedDate.getFullYear() === today.getFullYear()) {
                        
                        const currentHour = today.getHours();
                        const currentMinute = today.getMinutes();
                        
                        if (hour < currentHour || (hour === currentHour && minute < currentMinute)) {
                            continue;
                        }
                    }
                    
                    const timeSlot = document.createElement('div');
                    timeSlot.className = 'time-slot text-center py-2 bg-white rounded cursor-pointer hover:bg-amber-500 hover:text-white transition';
                    
                    // Format time (09:00 instead of 9:00)
                    const formattedHour = hour.toString().padStart(2, '0');
                    const formattedMinute = minute.toString().padStart(2, '0');
                    const timeString = `${formattedHour}:${formattedMinute}`;
                    
                    timeSlot.textContent = timeString;
                    timeSlot.dataset.time = timeString;
                    
                    timeSlot.addEventListener('click', function() {
                        selectTime(timeString);
                    });
                    
                    timeSlotsContainer.appendChild(timeSlot);
                }
            }
        }
        
        function selectTime(time) {
            selectedTime = time;
            
            // Update UI
            document.querySelectorAll('.time-slot').forEach(el => {
                el.classList.remove('bg-amber-500', 'text-white');
            });
            
            const selectedTimeElement = document.querySelector(`.time-slot[data-time="${time}"]`);
            if (selectedTimeElement) {
                selectedTimeElement.classList.add('bg-amber-500', 'text-white');
            }
            
            // Update booking summary
            updateBookingSummary();
        }
        
        function updateBookingSummary() {
            const summaryElement = document.getElementById('booking-summary');
            
            if (selectedService && selectedBarber && selectedDate && selectedTime) {
                const formattedDate = selectedDate.toLocaleDateString('pt-BR', { 
                    weekday: 'long', 
                    day: 'numeric', 
                    month: 'long' 
                });
                
                summaryElement.innerHTML = `
                    <strong>${selectedService}</strong> com <strong>${selectedBarber}</strong><br>
                    ${formattedDate} às ${selectedTime}
                `;
            } else if (selectedService && selectedBarber) {
                summaryElement.innerHTML = `Falta selecionar data e horário`;
            } else if (selectedService) {
                summaryElement.innerHTML = `Falta selecionar barbeiro, data e horário`;
            } else {
                summaryElement.innerHTML = `Selecione serviço, barbeiro e horário`;
            }
        }
        
        // Initialize calendar
        renderCalendar();
        
        // Month navigation
        document.getElementById('prev-month').addEventListener('click', function() {
            currentDate.setMonth(currentDate.getMonth() - 1);
            renderCalendar();
        });
        
        document.getElementById('next-month').addEventListener('click', function() {
            currentDate.setMonth(currentDate.getMonth() + 1);
            renderCalendar();
        });
        
        // Service selection
        document.querySelectorAll('.service-option').forEach(option => {
            option.addEventListener('click', function() {
                // Update UI
                document.querySelectorAll('.service-option').forEach(el => {
                    el.classList.remove('border-amber-500', 'bg-amber-50');
                });
                
                this.classList.add('border-amber-500', 'bg-amber-50');
                
                // Set selected service
                selectedService = this.dataset.service;
                
                // Update booking summary
                updateBookingSummary();
            });
        });
        
        // Barber selection
        document.querySelectorAll('.barber-option').forEach(option => {
            option.addEventListener('click', function() {
                // Update UI
                document.querySelectorAll('.barber-option').forEach(el => {
                    el.classList.remove('border-amber-500', 'bg-amber-50');
                });
                
                this.classList.add('border-amber-500', 'bg-amber-50');
                
                // Set selected barber
                selectedBarber = this.dataset.barber;
                
                // Update booking summary
                updateBookingSummary();
            });
        });
        
        // Quick booking from service cards
        document.querySelectorAll('.book-service').forEach(button => {
            button.addEventListener('click', function() {
                const service = this.dataset.service;
                
                // Find and click the corresponding service option
                document.querySelectorAll('.service-option').forEach(option => {
                    if (option.dataset.service === service) {
                        option.click();
                    }
                });
                
                // Scroll to booking section
                document.getElementById('booking').scrollIntoView({
                    behavior: 'smooth'
                });
            });
        });
        
        // Form submission
        document.getElementById('booking-form').addEventListener('submit', function(e) {
            e.preventDefault();
            
            if (!selectedService || !selectedBarber || !selectedDate || !selectedTime) {
                alert('Por favor, selecione serviço, barbeiro, data e horário.');
                return;
            }
            
            const name = document.getElementById('name').value;
            const phone = document.getElementById('phone').value;
            const email = document.getElementById('email').value;
            const notes = document.getElementById('notes').value;
            
            // Format date for display
            const formattedDate = selectedDate.toLocaleDateString('pt-BR', { 
                weekday: 'long', 
                day: 'numeric', 
                month: 'long',
                year: 'numeric'
            });
            
            // Show confirmation modal
            const modal = document.getElementById('booking-modal');
            const bookingDetails = document.getElementById('booking-details');
            
            bookingDetails.innerHTML = `
                <div class="text-left bg-gray-100 p-4 rounded-lg my-4">
                    <p><strong>Cliente:</strong> ${name}</p>
                    <p><strong>Serviço:</strong> ${selectedService}</p>
                    <p><strong>Barbeiro:</strong> ${selectedBarber}</p>
                    <p><strong>Data:</strong> ${formattedDate} às ${selectedTime}</p>
                    <p><strong>Contato:</strong> ${phone} | ${email}</p>
                    ${notes ? `<p><strong>Observações:</strong> ${notes}</p>` : ''}
                </div>
            `;
            
            modal.classList.remove('hidden');
            
            // Reset form
            this.reset();
            selectedService = null;
            selectedBarber = null;
            selectedDate = null;
            selectedTime = null;
            updateBookingSummary();
            
            // Reset UI selections
            document.querySelectorAll('.service-option, .barber-option').forEach(el => {
                el.classList.remove('border-amber-500', 'bg-amber-50');
            });
            
            document.querySelectorAll('.calendar-day, .time-slot').forEach(el => {
                el.classList.remove('bg-amber-500', 'text-white');
            });
        });
        
        // Close modal
        document.getElementById('close-modal').addEventListener('click', function() {
            document.getElementById('booking-modal').classList.add('hidden');
        });
    </script>
